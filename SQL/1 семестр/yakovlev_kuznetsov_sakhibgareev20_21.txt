CREATE TABLE sampling (
  id_measurements INT AUTO_INCREMENT PRIMARY KEY,
  designer_id INT,
  client_id INT,
  measurements TEXT,
  estimate_cost INT,
  status_ TINYINT(1),
  FOREIGN KEY (designer_id) REFERENCES designers(id_designer),
  FOREIGN KEY (client_id) REFERENCES clients(id_client)
);

select clients.name, clients.surname, sampling.measurements, sampling.estimate_cost
from clients
join sampling on clients.id_client = sampling.client_id;

select designers.surname, count(sampling.id_measurements) as orders_count
from designers
left join sampling on designers.id_designer = sampling.designer_id
group by designers.surname;

select production.materials, production.end_date, clients.surname, clients.address
from production
join prepayment on production.prepayment_id = prepayment.id_prepayment
join sampling on prepayment.sampling_id = sampling.id_measurements
join clients on sampling.client_id = clients.id_client;

select delivery.delivery_date, delivery.delivery_cost, clients.address
from delivery
join sampling on delivery.sampling_id = sampling.id_measurements
join clients on sampling.client_id = clients.id_client
where delivery.assembled = 1;

select * from sampling
order by estimate_cost desc;

select name, surname
from clients
order by surname asc, name asc;

select designer_id, count(*) as project_count
from sampling
group by designer_id;

select sampling_id, sum(amount) as total_prepayment
from prepayment
group by sampling_id
having total_prepayment > 45000;

select status_, avg(estimate_cost) as average_cost
from sampling
group by status_;
