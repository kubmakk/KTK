


create table client (
id int auto_increment primary key,
phone varchar(45),
balance decimal(10,2),
reg_date datetime,
login varchar(45),
clientcol varchar(45),

CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);

create table employee (
id int auto_increment primary key,
position varchar(45),
rights varchar(45),
login varchar(45),
full_name varchar(255),
full_lastname varchar(255),
full_patronymic varchar(45),
CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);

create table pc (
id int auto_increment primary key,
number int,
status varchar(45),
zone varchar(45),
configure text,
CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);

create table service (
id int auto_increment primary key,
name varchar(255),
price decimal(10,2),
type varchar(45),
CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);

create table sessie (
id int auto_increment primary key,
start_time datetime,
end_time datetime,
cost decimal(10,2),
client_id int,
pc_id int,
employee_id int,
foreign key (client_id) references client(id),
foreign key (pc_id) references pc(id),
foreign key (employee_id) references employee(id),
CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);

create table purshes (
id int auto_increment primary key,
description varchar(255),
service_id int,
client_id int,
employee_id int,
foreign key (service_id) references service(id),
foreign key (client_id) references client(id),
foreign key (employee_id) references employee(id),
CreatedAt DATETIME,
CreatedBy VARCHAR(50),
UpdatedAt DATETIME,
UpdatedBy VARCHAR(50),
IsActive BOOLEAN TRUE
);
